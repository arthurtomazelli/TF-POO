package ui;

import javax.swing.*;

public class CadastrarTecnologia extends JFrame implements ActionListener {
    private JPanel painelBorda;
    private JPanel painelPrincipal;
    private final List<String> stringLabels = new ArrayList<>(Arrays.asList("ID: ", "Modelo: ", "Descrição: ", "Valor Base: ", "Peso: ", "Temperatura: "));
    private List<JTextField> camposTexto;

    public CadastrarTecnologia() {
        super();
        setBasics();

        painelBorda = new JPanel(new BorderLayout());
        setBorda(painelBorda);

        painelPrincipal = new JPanel();
        painelPrincipal.setLayout(new BoxLayout(painelPrincipal, BoxLayout.Y_AXIS));

        painelPrincipal.add(criarPainelTitulo());
        painelPrincipal.add(criarPainelCampos());

        painelBorda.add(painelPrincipal);

        this.add(painelBorda);
        this.setVisible(true);

    }

    private void setBasics() {
        this.setTitle("Cadastrar Tecnologia");
        this.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        this.setSize(1200, 800);
        this.setLocationRelativeTo(null);
    }

    /*private void setBorda(JPanel painelBorda) {

    }*/

    private void setBorda(JPanel painelBorda) {
        painelBorda.setBorder(BorderFactory.createEmptyBorder(80, 100, 80, 100));

        this.addComponentListener(new ComponentAdapter() {
            @Override
            public void componentResized(ComponentEvent e) {
                int width = getWidth();
                int height = getHeight();

                int topBottom = (int) (height * 0.2);
                int leftRight = (int) (width * 0.22);

                painelBorda.setBorder(BorderFactory.createEmptyBorder(topBottom, leftRight, topBottom, leftRight));

                painelBorda.revalidate();
                painelBorda.repaint();
            }
        });
    }

    private JPanel criarPainelTitulo() {
        JPanel painelTitulo = new JPanel(new FlowLayout(FlowLayout.CENTER));
        painelTitulo.setBackground(new Color(20, 86, 160));

        JLabel titulo = new JLabel("CADASTRAR TECNOLOGIA");
        titulo.setFont(new Font("Arial", Font.BOLD, 40));
        titulo.setForeground(Color.WHITE);

        painelTitulo.add(titulo);

        return painelTitulo;
    }

    private JPanel criarPainelCampos() {
        JPanel painelGrid = new JPanel(new GridLayout(6, 2, 10, 10));

        camposTexto = new ArrayList<>();

        for(String s : stringLabels) {
            JTextField campo = new JTextField(50);
            JLabel atributo = new JLabel(s);
            Font fonteTexto = new Font("Arial", Font.BOLD, 30);

            atributo.setFont(fonteTexto);
            campo.setFont(fonteTexto);

            painelGrid.add(atributo);
            painelGrid.add(campo);
            camposTexto.add(campo);
        }

        return painelGrid;
    }

    @Override
    public void actionPerformed(ActionEvent e) {

    }
}
